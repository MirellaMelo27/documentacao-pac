FUNÇÃO: motorControl()

// ============================================================================== // FUNÇÃO: motorControl() // DESCRIÇÃO: Controla a velocidade e direção dos motores esquerdo e direito // PARÂMETROS: // - speedLeft: velocidade do motor esquerdo (-255 a 255) // valores negativos = sentido reverso // - speedRight: velocidade do motor direito (-255 a 255) // valores negativos = sentido reverso // RETORNO: void // ============================================================================== void motorControl(int speedLeft, int speedRight) { analogWrite(pinMotorLeft, abs(speedLeft)); analogWrite(pinMotorRight, abs(speedRight)); }

// ============================================================================== // FUNÇÃO: readDistance() // DESCRIÇÃO: Mede a distância usando o sensor ultrassônico HC-SR04 // PARÂMETROS: nenhum // RETORNO: float (distância em centímetros) // ============================================================================== float readDistance() { digitalWrite(trigPin, LOW); delayMicroseconds(2); digitalWrite(trigPin, HIGH); delayMicroseconds(10); digitalWrite(trigPin, LOW); long duration = pulseIn(echoPin, HIGH); return duration * 0.034 / 2; }
